### Linked List

```python
class Node:
    def __init__(self,value):
        self.value = value
        self.next = None

class LinkedList:
    def __init__(self):
        self.head = None

    def __len__(self):
        current = self.head
        count = 0
        while current:
            count += 1
            current = current.next
        return count

    def append(self,node):
        if self.head == None:
            self.head = node
            return
        current = self.head
        while current.next != None:
            current = current.next
        current.next = node

    def addFirst(self,node):
        current = self.head
        node.next = current
        self.head = node

    def clear(self):
        self.head = None

    def remove(self,val):
        dummyNode = Node("dummy")
        dummyNode.next = self.head
        self.head  = dummyNode
        current = self.head
        nextVal = current.next

        while nextVal:
            if nextVal.value == val:
                current.next = nextVal.next
            current = current.next
            nextVal = nextVal.next

        self.head = self.head.next


    def contains(self,num):
        current = self.head
        while current:
            if current.value == num:
                return True
            current = current.next
        return False

    def pop(self):
        current = self.head
        if current == None:
            return
        if current.next == None:
            self.head = None
            return
        while current.next.next != None:
            current = current.next
        current.next = None



    def __str__(self):
        current = self.head
        linkedList = ""
        while current:
            linkedList += f"{current.value} -> "
            current = current.next
        return linkedList
```

### Floyd's Tortoise and Hare AKA Cycle Detection Algorithm

- moving two pointers at different speeds through the sequence of values until they both point to equal values.
- Useful when detecting a cycle in a linked list.

```python
def hasCycle(self, head: Optional[ListNode]) -> bool:
    slow,fast = head,head
    while fast and fast.next:
        slow = slow.next
        fast = fast.next.next
        if slow == fast:
            return True
    return False
```
